- Create a repository r1

svcadmin create c:\svn\r1

- create a project r1-p1 - some patterns are used
svn mkdir file:///c:/svn/r1/r1-p1 -m "Create project p1"

- create 3 specific directories
cd temp
mkdir r1-p1
mkdir r1-p1\branches
mkdir r1-p1\tags
mkdir r1-p1\trunk
cd r1-p1
svn import file:///c:/svn/r1/r1-p1
rmdir /q /s c:\temp\r1-p1

go to the directory where the project is:
c:
svn co file:///c:/svn/r1/r1-p1
cd r1-p1
svn log -v
